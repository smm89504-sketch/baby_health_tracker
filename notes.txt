ثم عدل القيم التالية:
[mail function]
SMTP = smtp.gmail.com        
smtp_port = 587
sendmail_from = your@domain.com
sendmail_path = "\"C:\xampp\sendmail\sendmail.exe\" -t"

يجب ازالة الفاصلة المنقوطة ; و تعديل القيم بحيث نضع الايميل الخاص بنا الذي سنستخدمه لارسال الاشعار للايميل

[sendmail]

smtp_server=smtp.gmail.com
smtp_port=587
smtp_ssl=tls

auth_username=youremail@gmail.com
auth_password=your_app_password

default_domain=gmail.com

error_logfile=error.log
debug_logfile=debug.log

force_sender=youremail@gmail.com
